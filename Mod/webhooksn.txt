  <textarea id="mensagem" rows="4" cols="50"></textarea><br>
  <button onclick="enviarMensagem()">Enviar</button>

  <script>
    function enviarMensagem() {
      var mensagem = document.getElementById('mensagem').value;

      // Construir o objeto com os dados da mensagem
      var dados = {
        content: mensagem
      };

      // Enviar a mensagem para o webhook
      fetch('https://discord.com/api/webhooks/1124518204019056660/ZFuiu69QVg19Q26HXKJb8c7VyDxSmcEL5VFxIFwESveyn5NPhRTlEAPekcCg7aXjOGLT', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(dados)
      })
      .then(response => {
        if (response.ok) {
          console.log('Mensagem enviada com sucesso!');
        } else {
          console.error('Erro ao enviar mensagem:', response.status, response.statusText);
        }
      })
      .catch(error => {
        console.error('Erro ao enviar mensagem:', error);
      });
    }
  </script>
</body>
