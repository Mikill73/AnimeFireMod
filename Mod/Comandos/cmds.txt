const url = 'https://raw.githubusercontent.com/Mikill73/AnimeFireMod/main/Mod/Comandos/';

fetch(url)
  .then(response => response.text())
  .then(data => {
    const fileNames = data.match(/[^/]+$/gm); // Extrair nomes de arquivos
    if (fileNames) {
      alert(`Nomes dos arquivos:\n${fileNames.join('\n')}`);
    } else {
      alert('Nenhum arquivo encontrado.');
    }
  })
  .catch(error => {
    console.error('Erro:', error);
  });
